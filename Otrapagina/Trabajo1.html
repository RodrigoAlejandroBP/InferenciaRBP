Obtener la mediana

mediana <- function(x) {
  ordenar <- sort(x)
  z <- length(ordenar)
  if((z%%2) == 0) {              #Compruebo si es par o impar
    resultado <- ((x[z/2]+x[(z/2)+1])/2)
  }
  else{
    y <- (z+1)/2
    resultado <- ordenar[c(y)]
  }
  
  return(resultado)
}
Ejemplo impar
impar <- c(4,5,6)

## [1] 5

## [1] 5.5

Obtener la varianza

funcaux <- function(x){
  aux <- 0
  for(i in 1:length(x)){
    aux <- aux + x[i]/length(x)
  }
  return (aux)
}

varianza <- function(vector){
  aux <- 0
  for(i in 1:length(vector)){
    aux <- aux + ((vector[i] - funcaux(vector))^2)
  }
  
  return (aux/(length(vector)-1) )
}
Ejemplo de varianza con un vector de largo cuatro
ejemploVarianza<-c(2,2,1,1) # Se crea el nuevo vector

varianza(ejemploVarianza)
## [1] 0.3333333
Ejemplo de varianza con un vector de largo cuatro usando la funcion predefinida

Percentiles

perc50 <-function(x){
  alpha <- 0
  ordenada <- sort(x)
  alpha <- 1/2
  n <- length(ordenada)
  i <- alpha*(n+1)
  return((1-0)*ordenada[i]+0*ordenada[i+1])
  }
  
perc50(c(168,173,175,178,179,180,187))
## [1] 178
